local model = models.model.Skull

local cursorPos = vec(0, 0)
local screenSize = vec(10, 7)

local lastUpdateTime = 10

local oldViewerRot = vec(0, 0)

local function render(delta)
   local viewer = client.getViewer()
   if not viewer:isLoaded() then return end
   local rot = viewer:getRot()
   if lastUpdateTime > 2 then
      oldViewerRot = rot
   end
   local rotDiff = rot - oldViewerRot
   rotDiff = rotDiff.yx / screenSize * 0.1
   cursorPos = cursorPos + rotDiff --[[@as Vector2]]
   cursorPos.x = math.clamp(cursorPos.x, 0, 1)
   cursorPos.y = math.clamp(cursorPos.y, 0, 1)
   oldViewerRot = rot
end

local renderUpdate = false
function events.world_render()
   renderUpdate = true
   lastUpdateTime = lastUpdateTime + 1
end

local renderOffsets = {
   [false] = vec(3.99, 0, 0),
   [true] = vec(-13.95, 0, 0),
}

local renderLeftHand = false
events.SKULL_RENDER:register(function(delta, block, item, entity, ctx)
   if renderUpdate then
      render(delta)
      lastUpdateTime = 0
      renderUpdate = false
   end
   if block then
      model:setPos(0, 4, 0)
      return
   end
   if not entity then
      model:setPos(0, 5, 0)
      return
   end
   if ctx:match("FIRST_PERSON") then
      renderLeftHand = false
      if ctx:match("LEFT") then
         renderLeftHand = true
      end
   elseif ctx:match("THIRD_PERSON") then
      model:setPos(0, 8, -2)
      return
   end
   local pos = vec(0, 12.85, 0) + renderOffsets[renderLeftHand]
   pos.xy = pos.xy + screenSize * cursorPos
   pos.z = pos.z
   model:setPos(pos)
end)
